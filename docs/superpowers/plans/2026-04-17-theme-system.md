# Theme System Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a full Shiki-powered theme system — CSS custom properties per theme, JetBrains Mono font site-wide, theme picker in the sliding panel, no hardcoded colours anywhere.

**Architecture:** A custom Astro integration (`src/theme-integration.mjs`) runs in `astro:config:setup` and extracts colours from all bundled Shiki themes, writing `public/themes.css` (one `:root[data-theme]` block per theme) and `src/generated/theme-list.json` (name + swatch data for the panel UI). `BaseLayout.astro` links the CSS and applies the stored theme via an inline `<script>` before first paint. `SlidingPanel.astro` reads the JSON at build time and renders a scrollable list of themed rows; clicking one sets `data-theme` on `<html>` and persists to `localStorage`.

**Tech Stack:** Astro 6.x, shiki (explicit dev dep), @fontsource-variable/jetbrains-mono, Tailwind CSS v4

## Task 2: Write the theme integration

**Files:**
- Create: `src/theme-integration.mjs`

- [ ] **Step 1: Create `src/theme-integration.mjs`**

```js
import { writeFileSync, mkdirSync } from 'node:fs'
import { resolve, dirname } from 'node:path'
import { fileURLToPath } from 'node:url'

const __dirname = dirname(fileURLToPath(import.meta.url))

/**
 * Search tokenColors for the first rule matching any of the given TextMate scopes.
 * Returns the foreground hex string, or null.
 */
function extractTokenColor(tokenColors, targetScopes) {
  for (const rule of tokenColors) {
    const scopes = Array.isArray(rule.scope)
      ? rule.scope
      : typeof rule.scope === 'string'
        ? [rule.scope]
        : []
    const hit = scopes.some(s =>
      targetScopes.some(t => s === t || s.startsWith(t + '.'))
    )
    if (hit && rule.settings?.foreground) return rule.settings.foreground
  }
  return null
}

export function themeIntegration() {
  return {
    name: 'halcyon-theme-integration',
    hooks: {
      'astro:config:setup': async ({ logger }) => {
        logger.info('Generating theme CSS and theme list...')

        const { bundledThemes } = await import('shiki')

        const themeList = []
        const cssBlocks = []

        for (const [name, loader] of Object.entries(bundledThemes)) {
          try {
            const { default: theme } = await loader()
            // Some themes use `settings` (old TextMate format), others `tokenColors`
            const tokenColors = theme.tokenColors || theme.settings || []

            const bg = theme.bg
              || theme.colors?.['editor.background']
              || '#1a1a1a'
            const fg = theme.fg
              || theme.colors?.['editor.foreground']
              || '#f8f8f2'

            const accent = extractTokenColor(tokenColors, [
              'keyword', 'storage.type', 'keyword.control',
              'entity.name.function', 'support.function',
            ]) || fg

            const accent2 = extractTokenColor(tokenColors, [
              'string', 'string.quoted.double',
              'string.quoted.single', 'string.template',
            ]) || accent

            const muted = extractTokenColor(tokenColors, [
              'comment', 'comment.line',
              'comment.block', 'comment.block.documentation',
            ]) || fg

            // swatch: three distinct colours for the panel preview dots
            themeList.push({ name, bg, swatch: [accent, accent2, muted] })

            cssBlocks.push(
              `:root[data-theme="${name}"] {\n` +
              `  --bg: ${bg};\n` +
              `  --fg: ${fg};\n` +
              `  --accent: ${accent};\n` +
              `  --accent-2: ${accent2};\n` +
              `  --muted: ${muted};\n` +
              `}`
            )
          } catch (e) {
            logger.warn(`[theme-integration] Skipping "${name}": ${e.message}`)
          }
        }

        // Write public/themes.css
        const publicDir = resolve(__dirname, '..', 'public')
        writeFileSync(
          resolve(publicDir, 'themes.css'),
          '/* Auto-generated by theme-integration.mjs — do not edit */\n\n' +
          cssBlocks.join('\n\n') + '\n'
        )

        // Write src/generated/theme-list.json
        const generatedDir = resolve(__dirname, 'generated')
        mkdirSync(generatedDir, { recursive: true })
        writeFileSync(
          resolve(generatedDir, 'theme-list.json'),
          JSON.stringify(themeList, null, 2) + '\n'
        )

        logger.info(`[theme-integration] Generated ${themeList.length} themes`)
      },
    },
  }
}
```

  const DEFAULT_THEME = 'synthwave-84'
  const panel = document.getElementById('slidingPanel')
  const toggleBtn = document.getElementById('toggleButton')
  const closeBtn = document.getElementById('panelClose')
  let isOpen = false

  function markActive(currentTheme) {
    document.querySelectorAll('[data-theme-name]').forEach(function(btn) {
      var active = btn.dataset.themeName === currentTheme
      btn.style.outline = active ? '2px solid var(--accent)' : 'none'
      btn.style.outlineOffset = active ? '1px' : '0'
    })
  }

  function openPanel() {
    panel.style.width = '240px'
    isOpen = true
    markActive(localStorage.getItem('theme') || DEFAULT_THEME)
  }

  function closePanel() {
    panel.style.width = '0'
    isOpen = false
  }

  toggleBtn.addEventListener('click', function() { isOpen ? closePanel() : openPanel() })
  closeBtn.addEventListener('click', closePanel)

  document.querySelectorAll('[data-theme-name]').forEach(function(btn) {
    btn.addEventListener('click', function() {
      var name = btn.dataset.themeName
      document.documentElement.setAttribute('data-theme', name)
      localStorage.setItem('theme', name)
      markActive(name)
      closePanel()
    })
  })
</script>
